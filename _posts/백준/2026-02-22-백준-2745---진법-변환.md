```python

# A = 아스키 65, Z = 90
# 즉 ord('A') = 65
# -55를 해 주어야 함.

# ZZZZZ 36이면?
# 36^0 * Z + 36^1 * Z + ...
# 답은? 



a, b = input().split()
# a = 수, b = 진법

sum = 0
p = len(a) - 1
for i in range(len(a)):
    if a[i].isdigit():
        sum += (ord(a[i]) - 48) * (int(b) ** p)
    else:
        sum += (ord(a[i]) - 55) * (int(b) ** p)
    p -= 1
print(sum)
```
